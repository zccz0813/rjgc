
# 1. 初始化本地仓库
git init

# 2. 添加当前文件夹下所有文件
git add .

# 3. 提交到本地仓库
git commit -m "first commit"

# 4. 设置主分支名
git branch -M main

# 5. 连接到你 GitHub 上的仓库
git remote add origin https://github.com/xxxx/xxxx.git

# 6. 推送代码到 GitHub
git push -u origin main

# 先查看一下当前有哪些文件要被提交
git status

# 提交所有已添加的文件
